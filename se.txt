#script pipe

pipeline {
    agent any

    tools {
        maven 'MAVEN-HOME'
    }

    stages {
        stage('git repo & clean') {
            steps {
                bat 'git clone https://github.com/akshayabelde/maven.git'
                bat 'mvn clean -f maven/pom.xml'
            }
        }

        stage('install') {
            steps {
                bat 'mvn install -f maven/pom.xml'
            }
        }

        stage('test') {
            steps {
                bat 'mvn test -f maven/pom.xml'
            }
        }

        stage('package') {
            steps {
                bat 'mvn package -f maven/pom.xml'
            }
        }
    }
}
